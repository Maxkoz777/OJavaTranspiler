class MathUtils is

  var min : Integer
  var max : Integer
  function<Integer, Integer> f := a -> Integer.Max

  this(bt : Integer, ar : Integer) is
      max := bt
      min := ar
  end

  method MaxInt(a: Integer) : Integer is
   var m : Integer.Min
   var i : Integer.get(1)
   while i > m loop
      if a > m then m := a.get(i) + i end
      i := i + 1
      m := 3
      m := this.f(min) + i
   end
   return max + i
  end

  end

  method isPrime(k2: Integer) : Boolean is

    j := 2
