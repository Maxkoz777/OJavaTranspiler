class MathUtils is
  var intMathMin : Integer
  var intMathMax : Integer

  this(bt : Integer, ar : Integer) is
      intMathMax := bt
      intMathMin := ar
  end

  method empty(btw : Integer, arw : Integer) : Integer is
  end

  method intMathMaxInt(mathA: Integer) : Integer is
   var m : Integer.intMathMin
   var mathI : Integer.get(1)
   while mathI > m loop
   if mathA > m then m := mathA.get(mathI) + mathI end
   mathI := mathI + 1
   m := 3
   end
   return intMathMax + mathI
  end

  class Child is
  end

  method isPrime() : Boolean is
    var mathJ : Integer
    mathJ := 2
    return mathJ - 1
  end
end
